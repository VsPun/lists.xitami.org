Thats the back-end VB program:

Sub main()
     Open "c:\test.txt" For Output Access Write As #1
     Print #1, "start"
     a$ = Environ$("CGI_STDOUT")
     Open a$ For Output Access Write As #3
     Print #3, "<html><body><pre>"
     a$ = Environ$(1)
     While a$ <> ""
         Print #3, a$; "<br>"
         Print #1, a$
         i = i + 1
         a$ = Environ$(i)
     Wend
     Print #3, "</body></html>"
     Close
     End
End Sub
