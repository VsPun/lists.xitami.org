[Cgi]
    Debug = "1"
    Exit-ok = "0"
    Form-query = "1"
    Full-name = "1"
    Timeout = "120"
    Wildcard = "1"
[Filter]
    .php = "c:\\php\\php.exe"
    .phps = "C:\\php\\php.exe -s"
    .phtml = "C:\\php\\php.exe"
    .shtm = "perlssi"
    .shtml = "perlssi"
    .ssi = "perlssi"

[Ssi]
    Enabled = "1"
    Exec = "1"
[Win32]
    16bit-cgi = "1"
    Service-name = "Xitami"
    Service-text = "Xitami Web Server"

Ok now you can add to the Autoexec.bat
SET PATH=F:\PERL\BIN;%PATH%

Next check your
cgierr.log file that is in the xitami directory

Xitami on windows look for Peal in the PATH;
