You may also need this in your defaults.cfg file

[Cgi]
    Debug = "1"
    Exit-ok = "0"
    Form-query = "1"
    Full-name = "1"
    Timeout = "120"
    Wildcard = "1"

[Filter]
    . = "text/plain"
    .pl = "/usr/local/bin/perl"
    .php = "/usr/local/bin/php"
    .shtm = "perlssi"
    .shtml = "perlssi"
    .ssi = "perlssi"


hope it helps
