{ my ($to, $from, $subject, @body, $attach)=@_;

  my $relay = "smtp.earthlink.net";

  die "Open of mail failed" if (! defined 
$smtp);
  print "Open of mail OK", " BR ";

 

  print "Mail beg", " BR ";
  
$smtp- mail($from);
  $smtp- to($to);
